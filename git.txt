//Создаем новую и переключаемся в новую ветку
git checkout -b "название ветки"

// Добавить файлы в git
git add .

//Выполняем коммит в локальный git
git commit -m "***пишем коммент ***"

//Публикуем ветку в git
git push -u origin имя_ветки

// перейти в основную ветку из отдельной ветки
git checkout master

// получить обновления по активной ветке
git pull
